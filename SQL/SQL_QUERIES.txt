SQL> --22. Build a CTE to calculate monthly sales and find above-average months.
SQL> 
SQL> WITH monthly_sales AS (
  2      SELECT
  3          TO_CHAR(o.order_date, 'YYYY-MM') AS month,
  4          SUM(oi.quantity * p.price) AS total_sales
  5      FROM ORDERS o
  6      JOIN ORDER_ITEMS oi ON o.order_id = oi.order_id
  7      JOIN PRODUCTS p ON oi.product_id = p.product_id
  8      GROUP BY TO_CHAR(o.order_date, 'YYYY-MM')
  9  )
 10  SELECT *
 11  FROM monthly_sales
 12  WHERE total_sales > (
 13      SELECT AVG(total_sales) FROM monthly_sales
 14  );

  2      SELECT
  3          o.customer_id,
  4          COUNT(DISTINCT o.order_id) AS total_orders,
  5          SUM(oi.quantity * p.price) AS total_order_value
  6      FROM ORDERS o
  7      JOIN ORDER_ITEMS oi ON o.order_id = oi.order_id
  8      JOIN PRODUCTS p ON oi.product_id = p.product_id
  9      GROUP BY o.customer_id
 10  ),

 12      SELECT
 13          o.customer_id,
 14          SUM(pay.amount) AS total_payment

 18  )
